var ballx = 300;
var bally = 300;
var ballSize = 40;
var score = 0;
var gameState = "L1";
var img, img2, img3;
var cursorImg;

function preload() {
// preload() runs once, it may make you wait
//  img = loadImage('backgroundg.jpg');  // cat.jpg needs to be next to this .js file
// you can link to an image on your github account
  img = loadImage('https://Punzoul.github.io/images/backgroundg.jpg');
  img2 = loadImage('http://Punzoul.github.io/images/starg.png');
  cursorImg = loadImage('http://Punzoul.github.io/images/firstocg.png');
}

function setup() {
  createCanvas(600,600);
  noCursor();
  textAlign(CENTER);
  textSize(20);
}//end of setup ===========

function draw(){
  background(img);
  if(gameState == "L1"){
  levelOne();
  }
  if(gameState == "L2"){
  levelTwo();
  }
  if(gameState == "L3"){
    levelThree();
  }
  if(gameState == "win"){
    winGame();
  }

  text("Score: " + score, width / 2, 40);
}//end of draw ========

function levelOne(){
    image(cursorImg, mouseX-30, mouseY-30, width/10, height/10);
  text("Level 1", width/2, height-20);
  fill(255,204,0);
  var distToBall = dist(ballx, bally, mouseX, mouseY);
  if(distToBall < ballSize/2){
    ballx = random(width);
    bally= random(height);
    score= score + 1;
  }
  
  if(score>= 5){
   gameState = "L2";
  }
  
  line(ballx, bally, mouseX, mouseY);
  image(img2, ballx-ballSize/2, bally-ballSize/2, ballSize, ballSize);

  
}//end of level 1========

function levelTwo(){
  background(img);
    image(cursorImg, mouseX-30, mouseY-30, width/10, height/10);
  text("Level 2", width/2, height-20);
   fill(255,204,0);
  var distToBall = dist(ballx, bally, mouseX, mouseY);
  if (distToBall 10){
   gameState = "L3";
  }
  //line(ballx, bally, mouseX, mouseY);
  image(img2, ballx-ballSize/2, bally-ballSize/2, ballSize, ballSize);

}//end of level 2========

function levelThree(){
  background(img);
    image(cursorImg, mouseX-30, mouseY-30, width/10, height/10);
  text("Level 2", width/2, height-20);
   fill(255,204,0);
  var distToBall = dist(ballx, bally, mouseX, mouseY);
  if (distToBall 15){
    gameState = "win";
    background(random(255));
  }
  //line(ballx, bally, mouseX, mouseY);
  image(img2, ballx-ballSize/2, bally-ballSize/2, ballSize, ballSize);

  
}//end of level 3========

}//end of winGame========
